134 template<typename ValueType,
135  FFNN_SIZE_TYPE InputsAtCompileTime,
136  FFNN_SIZE_TYPE OutputsAtCompileTime>
137 void SparselyConnected<ValueType, InputsAtCompileTime, OutputsAtCompileTime>::reset()
138 {
139  typedef Eigen::Matrix<ValueType, OutputsAtCompileTime, InputsAtCompileTime> RandMatrix;
140 
141  FFNN_ASSERT_MSG(Base::isInitialized(), "Layer is not initialized.");
142 
143  // Create random value matrix
144  RandMatrix random(Base::outputShape().size(), Base::inputShape().size());
145  random.setRandom(Base::outputShape().size(), Base::inputShape().size());
146 
147  // Build weight matrix
148  w_.resize(Base::outputShape().size(), Base::inputShape().size());
149  for (SizeType idx = 0; idx < Base::outputShape().size(); idx++)
150  {
151  for (SizeType jdx = 0; jdx < Base::inputShape().size(); jdx++)
152  {
153  const ValueType p = (random(idx, jdx) + 1) / 2;
154  if (p < config_.connection_probability)
155  {
156  w_.insert(idx, jdx) =
157  config_.init_weight_mean + random(idx, jdx) * config_.init_weight_std;
158  }
159  }
160  }
161 
162  // Set uniformly random bias matrix + add biases
163  b_.setRandom(Base::outputShape().size(), 1);
164  b_ *= config_.init_bias_std;
165  if (std::abs(config_.init_bias_mean) > 0)
166  {
167  b_.array() += config_.init_bias_mean;
168  }
169 }
